Use this code block to produce one publication quality plot of this data in a way that does a good job of illustrating the differences in gene expression in starved and well fed individuals as well as the wild caught individuals.

# This will give us a color vector by the condition
# remember factors are alphabetical so the order will be
# ad libitum, starved, wild
cols <- c(rgb(1, 0, 0, .6), 
          rgb(0, 0, 1, .6), 
          rgb(.1, .1, .1, .6))[as.factor(dat$condition)]

# this will give us shapes associated with sex
# order will be female male
pchs <- c(19, 15)[as.factor(dat$sexes)]

# basic plot
plot(x = gene.pca$x[, 1],
     y = gene.pca$x[, 2],
     pch = pchs,
     col = cols,
     cex = .75,
     xlim = c(-75, 60),
     xlab = "PC1",
     ylab = "PC2")
# legend points
points(x = rep(-76, 6),
       y = seq(from = 48, to = 30, length.out = 6),
       pch = c(15, 19, 15, 19, 15, 19),
       col = c(rgb(0, 0, 1, .6),
               rgb(0, 0, 1, .6),
               rgb(1, 0, 0, .6),
               rgb(1, 0, 0, .6),
               rgb(.1, .1, .1, .6),
               rgb(.1, .1, .1, .6)))
# legend text
text(x = rep(-76, 6),
     y = seq(from = 48, to = 30, length.out = 6),
     pos = 4,
     cex = .6,
     labels = c("male - starved", "female - starved",
                "male - ad libitum", "female - ad libitum",
